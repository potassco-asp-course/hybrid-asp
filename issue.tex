\begin{frame}{Founded versus external theory atoms}{One of many issues}
  \begin{center}
    ``\texttt{\&sum\{x;-y\} <= 4}'' \qquad\qquad $x-y\leq 4$
   \end{center}
   \bigskip
  \begin{itemize}
    \item<2-> \structure{Founded}  \ truth of a theory atom implies truth of its constraint
    \item<2-> \structure{External} \ truth of a theory atom is equivalent to its constraint
      \bigskip
    \item<3-> \structure{Example}
      \begin{itemize}
      \item \clingcon:  all theory atoms are external
      \item \clingo-dl: theory atoms in the head/body are founded/external
      \end{itemize}
      \medskip
    \item<3-> \structure{Aspiration} \ axiomatization of different approaches
  \end{itemize}
\end{frame}
