\begin{frame}[fragile]{Equivalence in \clingcon}
\begin{semiverbatim}

                     a :- &sum\{ x; y \} = 4.
      &sum\{ y; z \} = 2 :- a.\pause


                     a :- &sum\{ x; y \} = 4.
                       :- a, not &sum\{ y; z \} = 2.\pause


                     a :- &sum\{ x; y \} = 4.
                       :- a, &sum\{ y; z \} != 2.
\end{semiverbatim}
\end{frame}

\begin{frame}[fragile]{Equivalence \clingodl}
\begin{semiverbatim}

                 margin :- &diff\{ x-y \} <= 10.
      &diff\{ x-y \} <= 0 :- not margin.
      &diff\{ y-x \} <= 0 :- not margin.\pause


                 margin :- &diff\{ x-y \} <= 10.
                        :- &diff\{ y-x \} <= -1, not margin.
                        :- &diff\{ x-y \} <= -1, not margin.
\end{semiverbatim}
\end{frame}
